================POST================

curl -i -X POST http://localhost:8080/upload -F "file=@my_tests/small.txt"

curl -i -X POST http://localhost:8080/upload -F "file=@my_tests/image.jpg"

curl -i -X POST http://localhost:8080/upload -H "Content-Type: plain/text" --data "salut"

curl -i -v -X POST http://localhost:8080/upload -H "Transfer-Encoding: chunked" -F "file=@my_tests/small.txt"

curl -i -v -X POST http://localhost:8080/upload/ -H "Transfer-Encoding: chunked" --data-binary @- 

curl -i -v -X POST http://localhost:8080/upload/ -F "file=@my_tests/big.txt" --trace -
(413 direct car content_length connu)

curl -i -v -X POST http://localhost:8080/upload/ -H "Transfer-Encoding: chunked" -F "file=@my_tests/big.txt" --trace -
(413 des qu'on atteint la client_max_body_size)
